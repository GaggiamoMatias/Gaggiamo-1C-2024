225static void measureTask(void *pvParameter)
226{
227 while(true)
228 {
229 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
230
231 temp = Si7007MeasureTemperature();
232 hum = Si7007MeasureHumidity();
233 valor_gas = MQGetPercentage(MQRead()/RO_);
234
235 valor_gas_convertido = (uint16_t)(valor_gas*1000000);
236 temp_converted = (uint16_t)(temp);
237 hum_converted = (uint16_t)(hum);
238
239 UartSendString(UART_PC, "\nTemperatura: ");
240 UartSendString(UART_PC, (const char*)UartItoa(temp_converted, 10));
241 UartSendString(UART_PC, " °C\r\n");
242
243 UartSendString(UART_PC, "Humedad: ");
244 UartSendString(UART_PC, (const char*)UartItoa(hum_converted, 10));
245 UartSendString(UART_PC, "%\r\n");
246
247 UartSendString(UART_PC, "Concentracion de gas: ");
248 UartSendString(UART_PC, (const char*)UartItoa(valor_gas_convertido, 10));
249 UartSendString(UART_PC, " ppm.\r\n");
250
251 if(temp_converted >= umbral_temp || hum_converted >= umbral_hum || valor_gas_convertido >= umbral_gas)
252 {
253 activarAlarma = true;
254 permisoPuerta = false;
255 }
256 else
257 {
258 activarAlarma = false;
259 permisoPuerta = true;
260 }
261 }

397void handleRFIDCard()
398{
399 // show card UID
400 UartSendString(UART_PC,"\nCard uid bytes: ");
401 for (uint8_t i = 0; i < mfrcInstance->uid.size; i++)
402 {
403 UartSendString(UART_PC," 0X");
404 UartSendString(UART_PC, (char*)UartItoa(mfrcInstance->uid.uidByte[i], 16));
405 UartSendString(UART_PC," ");
406 }
407 UartSendString(UART_PC,"\n\r");
408
409 // Convert the uid bytes to an integer, byte[0] is the MSB
410 last_user_ID =
411 (int)mfrcInstance->uid.uidByte[3] |
412 (int)mfrcInstance->uid.uidByte[2] << 8 |
413 (int)mfrcInstance->uid.uidByte[1] << 16 |
414 (int)mfrcInstance->uid.uidByte[0] << 24;
415
416 UartSendString(UART_PC,"Card Read user ID: ");
417 UartSendString(UART_PC, (char*)UartItoa(last_user_ID, 10));
418 UartSendString(UART_PC,"\n\r");
419
420 // Validate the UID
421 if (validateUID(mfrcInstance->uid.uidByte, mfrcInstance->uid.size) && permisoPuerta)
422 {
423 UartSendString(UART_PC, "\nAccess Granted\n\r");
424 abrirPuerta = true; // Permitir acceso
425 }
426 else
427 {
428 UartSendString(UART_PC, "\nAccess Denied\n\r");
429 abrirPuerta = false; // Denegar acceso
430 }
431}
