152 .func_p = FuncTimerA,

157 // Configuración e inicialización del Timer B utilizado para la tarea de mostrar por pantalla y cambiar el estado de los LEDs
158 // a partir de la medición.
159 timer_config_t timer_show =
160 {
161 .timer = TIMER_B,
162 .period = CONFIG_PERIOD_SHOW,
163 .func_p = FuncTimerB,
164 .param_p = NULL
165 };
166 TimerInit(&timer_show);
167
168 // Inicialización de funciones que activarán una interrupción ante la llamada de uno de los switches, llamando a las funciones
169 // enviadas como parámetro por referencia. En el caso del switch_1, se llamará a la función para medir o dejar de medir. En el caso
170 // del switch_2 a la función para mantener o dejar de mantener el resultado.
171 SwitchActivInt(SWITCH_1, &cambiar_estado_medicion, NULL);
172 SwitchActivInt(SWITCH_2, &mantener_resultado, NULL);
173
174 // Creación de tareas para medir la distancia y mostrar la misma por la pantalla LCD y por el cambio de estado de los LEDs.
175 xTaskCreate(&MeasureTask, "Measure", 1024, NULL, 5, &measure_task_handle);
176 xTaskCreate(&ShowTask, "Show", 512, NULL, 5, &show_task_handle);
177
178 // Comienzo de las cuentas de ambos timers.
179 TimerStart(timer_measure.timer);
180 TimerStart(timer_show.timer);
181}
