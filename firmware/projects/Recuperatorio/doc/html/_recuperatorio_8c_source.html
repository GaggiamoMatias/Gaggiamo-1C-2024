87static void DistanciaTask(void *pvParameter)
88{
89 while(true)
90 {
91 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
92
93 distancia = HcSr04ReadDistanceInCentimeters();
94
95 if(distancia>8 && distancia<12)
96 {
97 medirTemperatura = true;
98 LedOn(LED_2);
99 }
100 else if(distancia<8)
101 {
102 medirTemperatura = false;
103 LedOn(LED_1);
104 }
105 else if(distancia>12)
106 {
107 medirTemperatura = false;
108 LedOn(LED_3);
109 if(distancia>140)
110 {
111 reiniciarMedicion = true;
112 }
113 }
114 }
115}

121static void TemperaturaTask(void *pvParameter)
122{
123 uint16_t data_mV = 0;
124 float data_temp = 0;
125 float sumatoria_temp = 0;
126 float prom_temp = 0;
127 uint16_t prom_temp_convertido = 0;
128 uint8_t n_ciclos = 1;
129
130 while(true)
131 {
132 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
133
134 if(medirTemperatura)
135 {
136 AnalogInputReadSingle(CH1, &data_mV);
137
138 // Interpolacion lineal con Y=Temperatura, X=data[mV], X0=0 mV, X1=3300 mV, Y0=20°C y Y1=50°C.
139 data_temp = (30/3300)*data_mV + 20;
140 sumatoria_temp = sumatoria_temp + data_temp;
141 n_ciclos = n_ciclos + 1;
142
143 if(n_ciclos == 10)
144 {
145 prom_temp = sumatoria_temp / n_ciclos;
146 prom_temp_convertido = (uint16_t)prom_temp;
147
148 UartSendString(UART_PC, (const char*)UartItoa(prom_temp_convertido, 10));
149 UartSendString(UART_PC, "°C persona a ");
150 UartSendString(UART_PC, (const char*)UartItoa(distancia, 10));
151 UartSendString(UART_PC, " cm");
152
153 sumatoria_temp = 0;
154 n_ciclos = 0;
155
156 if(prom_temp > 37.5)
157 {
158 encenderAlarma = true;
159 }
160 else
161 {
162 encenderAlarma = false;
163 }
164 }
165 }
166 if(reiniciarMedicion)
167 {
168 sumatoria_temp = 0;
169 n_ciclos = 0;
170 reiniciarMedicion = false;
171 }
172 }
173}
